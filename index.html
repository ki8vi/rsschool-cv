Find Highest Rank Number in an Array

 
function highestRank(arr) {

let out = []
let hsh = {}
for(let i = 0; i < arr.length; i++) {
    if(!hsh[arr[i]]) {
    hsh[arr[i]] = 1
    } else {
    hsh[arr[i]]++
    }
}
arr = Object.entries(hsh)
const maxVal = Object.values(hsh).sort((a, b) => b-a)[0]
for(let i = 0; i < arr.length; i++) {
    if(arr[i][1] === maxVal) {
    out.push(+arr[i][0])
    }
}
    return Math.max(...out)
}
